**Create `worker/src/middleware/sanitize.ts` — Input sanitization middleware (SEC-01):**

This Hono middleware runs on all POST/PUT/PATCH requests. It:
1. Parses the JSON body
2. Recursively walks all string values
3. Strips HTML tags (removes `<script>`, `<img onerror>`, etc.)
4. Rejects strings containing SQL injection patterns (`'; DROP`, `UNION SELECT`, `OR 1=1`)
5. Rejects strings containing prompt injection markers (`[INST]`, `<<SYS>>`, `ignore previous`, `system:`)
6. Returns 400 with `{ error: "Invalid input" }` if any string fails validation

```typescript
import type { Context, Next } from "hono";

const XSS_PATTERN = /<\s*script|<\s*img\s+[^>]*onerror|javascript:|on\w+\s*=/i;
const SQLI_PATTERN = /(';\s*(DROP|ALTER|DELETE|UPDATE|INSERT)|UNION\s+SELECT|\bOR\s+1\s*=\s*1)/i;
const PROMPT_INJECTION_PATTERN = /(\[INST\]|<<SYS>>|ignore\s+(all\s+)?previous|system\s*:|<\|im_start\|>)/i;

function containsMalicious(value: string): boolean {
  return XSS_PATTERN.test(value) || SQLI_PATTERN.test(value) || PROMPT_INJECTION_PATTERN.test(value);
}

function checkObject(obj: unknown): boolean {
  if (typeof obj === "string") return containsMalicious(obj);
  if (Array.isArray(obj)) return obj.some(checkObject);
  if (obj && typeof obj === "object") {
    return Object.values(obj).some(checkObject);
  }
  return false;
}

export async function sanitize(c: Context, next: Next) {
  if (["POST", "PUT", "PATCH"].includes(c.req.method)) {
    try {
      const body = await c.req.json();
      if (checkObject(body)) {
        return c.json({ error: "Invalid input" }, 400);
      }
    } catch {
      // No JSON body or parse error — let downstream handle
    }
  }
  await next();
}
```

**Create `worker/src/middleware/idor-check.ts` — IDOR enforcement utility (SEC-03):**

Every D1 query on user-owned data MUST use this utility. It returns null (meaning 403) if the userId does not own the record. Pattern from 01-RESEARCH.md.

```typescript
import { eq, and } from "drizzle-orm";
import type { DrizzleD1Database } from "drizzle-orm/d1";

export async function verifyOwnership<T extends Record<string, unknown>>(
  db: DrizzleD1Database,
  table: any,
  recordId: string,
  userId: string,
  idCol: any,
  ownerCol: any,
): Promise<T | null> {
  const result = await db
    .select()
    .from(table)
    .where(and(eq(idCol, recordId), eq(ownerCol, userId)))
    .limit(1);
  return (result[0] as T) ?? null;
}
```
